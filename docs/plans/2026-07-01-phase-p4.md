- Hiding cross-boundary deps to make a "clean" C4 (violates SPEC §4 honesty) — Task 2 K-edge
  test + negative test guard this.
- Cutting clusters out of a sequence to force isolation (violates §4 PATH threading) — Task 4
  multi-cluster test guards this.
- Rendering React Flow for C4/sequence (conflicts with ADR-001 — RF is the graph renderer only).
- Owning the tab chrome from P4 (must defer to P3's `PaneTabs`; P4's role is CONTENT via
  `RepresentationContent` — see contract §2).
- LOD-truncating `crossBoundaryDeps` (compromises §4 honesty under PERF-005 cap) — Task 10
  test asserts the dep list stays complete under `maxContainers = 50`.

```typescript
import type { TypedGraphModel, AtlasEdge, NodeRole } from './typedGraphModel';
export interface C4Container { id: string; label: string; role: NodeRole; nodeIds: string[]; }
export interface C4Boundary { id: string; label: string; containerIds: string[]; }
export interface CrossBoundaryDep { from: string; to: string; toLabel: string; relation: AtlasEdge['relation']; label: string; }
export interface C4Diagram { boundary: C4Boundary; containers: C4Container[]; internalEdges: AtlasEdge[]; crossBoundaryDeps: CrossBoundaryDep[]; }

export function deriveC4(model: TypedGraphModel, opts: { clusterId: number; maxContainers?: number }): C4Diagram {
  const inCluster = model.nodes.filter(n => n.community === opts.clusterId);
  const inIds = new Set(inCluster.map(n => n.id));
  const containers = inCluster.map(n => ({ id: n.id, label: n.label, role: n.role, nodeIds: [n.id] }));
  const internalEdges = model.edges.filter(e => inIds.has(e.source) && inIds.has(e.target));
  return { boundary: { id: `cluster-${opts.clusterId}`, label: `cluster ${opts.clusterId}`, containerIds: containers.map(c => c.id) }, containers, internalEdges, crossBoundaryDeps: [] };
}
```
